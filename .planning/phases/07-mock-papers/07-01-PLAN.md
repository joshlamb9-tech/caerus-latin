```javascript
function toggleAnswers(section) {
  var el = document.getElementById('answers-' + section);
  if (!el) return;
  var isShown = el.classList.toggle('shown');
  var btn = el.previousElementSibling;
  if (btn && btn.classList.contains('show-answers-btn')) {
    btn.textContent = isShown ? 'Hide mark scheme for Section ' + section.toUpperCase() : 'Show mark scheme for Section ' + section.toUpperCase();
  }
}
```
