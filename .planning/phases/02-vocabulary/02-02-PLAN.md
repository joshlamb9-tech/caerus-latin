**Speed-recall mode:**

Timer implementation: `setInterval` at 1-second ticks decrementing a counter. On reveal: clear the interval, show answer for 1500ms via `setTimeout`, then advance.

**Filter logic (applyFilters function):**
```javascript
function applyFilters(words, params) {
  let result = words;
  if (params.get('gcse') === '1') result = result.filter(w => w.gcse_only === true);
  const pos = params.get('pos');
  if (pos === 'other') {
    result = result.filter(w => ['preposition','conjunction','numeral','pronoun'].includes(w.part_of_speech));
  } else if (pos) {
    result = result.filter(w => w.part_of_speech === pos);
  }
  return result;
}
```

**Entry point:**
```javascript
'use strict';
document.addEventListener('DOMContentLoaded', () => {
  const params = new URLSearchParams(window.location.search);
  const mode = params.get('mode');
  const app = document.getElementById('app');

  fetch('../data/vocabulary/all.json')
    .then(r => { if (!r.ok) throw new Error('vocab load failed'); return r.json(); })
    .then(allWords => {
      const words = applyFilters(allWords, params);
      if (mode === 'flashcard') return runFlashcard(app, words, params);
      if (mode === 'speed-recall') return runSpeedRecall(app, words, params);
      return renderBrowse(app, words, allWords, params);
    })
    .catch(() => {
      app.innerHTML = '<p class="error">Could not load vocabulary. Please reload.</p>';
    });
});
```
