Create the `gcse/` directory and `gcse/index.html`. The page has three display states managed by the auth guard:

**State A — Not logged in:** Redirect to `../login.html?next=gcse/index.html` (standard CE behaviour)

**State B — Logged in, no latin-gcse grant:** Show teaser overlay. Do NOT redirect. Sub-check for `latin` grant to determine copy variant:
- Has `latin` grant → "You have CE Latin access — upgrade to GCSE Latin for £49.99/year"
- Has no grant at all → "Unlock GCSE Latin — everything for OCR J282 — £49.99/year"

**State C — Logged in, has latin-gcse grant:** Remove auth-guard style, show subscriber hub.

**Auth guard modification for GCSE pages:**
Unlike CE pages, GCSE pages must NOT redirect users without a grant — they show a teaser instead. The guard logic is:
1. No session → redirect to login.html
2. Has session → always reveal page (remove auth-guard style), THEN check for latin-gcse grant
3. Has latin-gcse grant → set `window.CR_GCSE_ACCESS = true`, dispatch `cr-gcse-auth` event
4. No latin-gcse grant → set `window.CR_GCSE_ACCESS = false`, check for latin grant (for copy variant)

    (function () {
      var SUPABASE_URL  = 'https://dlcseuejvducbsjhqvze.supabase.co';
      var SUPABASE_ANON = 'sb_publishable_k_59EUPnHNnJ534tF0vMvQ_Ams9wV9L';
      var client = supabase.createClient(SUPABASE_URL, SUPABASE_ANON);

      client.auth.getSession().then(function (result) {
        var session = result.data.session;
        if (!session) {
          window.location.href = '../login.html?next=' + encodeURIComponent('gcse/index.html');
          return;
        }

        // Always reveal page — GCSE pages tease rather than redirect
        window.CR_AUTH = { client: client, userId: session.user.id };
        var guard = document.getElementById('auth-guard');
        if (guard) guard.remove();

        // Inject sign-out link
        var nav = document.getElementById('site-nav');
        if (nav) {
          var ul = nav.querySelector('ul');
          if (ul) {
            var li = document.createElement('li');
            var a  = document.createElement('a');
            a.href = '#';
            a.textContent = 'Sign out';
            a.style.opacity = '0.65';
            a.addEventListener('click', function (e) {
              e.preventDefault();
              client.auth.signOut().then(function () { window.location.href = '../login.html'; });
            });
            li.appendChild(a);
            ul.appendChild(li);
          }
        }

        // Check for latin-gcse grant
        client.from('access_grants')
          .select('id')
          .eq('user_id', session.user.id)
          .eq('product', 'latin-gcse')
          .gt('valid_until', new Date().toISOString())
          .limit(1)
          .then(function (gr) {
            if (gr.data && gr.data.length > 0) {
              // Subscriber hub view
              window.CR_GCSE_ACCESS = true;
              document.getElementById('gcse-teaser').style.display   = 'none';
              document.getElementById('gcse-hub').style.display      = 'block';
            } else {
              // Teaser view — check for CE subscription to customise copy
              window.CR_GCSE_ACCESS = false;
              document.getElementById('gcse-hub').style.display = 'none';
              client.from('access_grants')
                .select('id')
                .eq('user_id', session.user.id)
                .eq('product', 'latin')
                .gt('valid_until', new Date().toISOString())
                .limit(1)
                .then(function (ce) {
                  var hasCE = ce.data && ce.data.length > 0;
                  var teaser = document.getElementById('gcse-teaser');
                  teaser.style.display = 'block';
                  if (hasCE) {
                    document.getElementById('teaser-headline').textContent =
                      'You have CE Latin — upgrade to GCSE';
                    document.getElementById('teaser-sub').textContent =
                      'Add the full OCR J282 revision suite for £49.99/year.';
                  }
                });
            }
          });
      });
    })();
